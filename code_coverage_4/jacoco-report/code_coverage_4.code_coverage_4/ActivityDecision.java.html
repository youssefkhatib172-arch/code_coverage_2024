package code_coverage_4.code_coverage_4;

public class ActivityDecision {

    public static String decideActivity(
            boolean walking,
            boolean midnight,
            boolean running,
            boolean sunshine,
            boolean raining) {

        String result = "Keep doing what you were doing";

        if (walking && midnight) {
            if (raining) {
                result = "Take umbrella and searchlight";
            } else {
                result = "Take searchlight";
            }

        } else if (running && sunshine) {
            if (raining) {
                result = "Take umbrella";
            }
        }

        return result;
    }
}
