package code_coverage_4.code_coverage_4;

public class WeakMethods {

    // weakMethod: bug נחשף רק כשמריצים true-branch עם (a==0 && b==1)
    // ואפשר לעשות 100% Statement בלי להגיע לבאג (רק true-branch עם a!=0)
    public static int weakMethod(int a, int b) {
        int res = 1;

        // Decision: (a != 0) OR (b == 1)
        if (a != 0 || b == 1) {
            // תקלה: חלוקה ב-0 כאשר a==0 ו-b==1
            res = 10 / a;
        }

        return res;
    }
}
